#Include "TextLib" as TL

Text _JSON_Es(Text _V) {
	if (TL::Length(_V) < 1) return _V;
	declare T = TL::Replace(TL::Replace(_V, "\"", "&&q;;"), "\\", "\\\\");
	if (TL::Length(T) > 4) return TL::Replace(T, "&&q;;", "\\\"");
	return T;
}

Text _JSON_Ec(Text[Text] _V) {
	if (_V.count == 0) return "";
	declare Text Result = "{";
	declare Index = 0;
	foreach (Name => Value in _V) {
		Result ^= "\""^Name^"\":"^Value;
		if (Index < _V.count - 1) Result ^= ",";
		Index += 1;
	}
	return Result ^ "}";
}

Text JSON_Stringify(Boolean _V) {
	if (_V) return "true";
	return "false";
}
